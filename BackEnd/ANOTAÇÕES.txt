src/
├── controllers/       # Controladores para lidar com as requisições HTTP
│   └── users.js       # Controlador específico para rotas relacionadas a usuários
├── models/            # Modelos do Sequelize (representação das tabelas do banco de dados)
│   └── jogador.js     # Modelo para a tabela Jogador
│   └── sala.js        # Modelo para a tabela Sala
├── routes/            # Definição de rotas
│   └── users.js       # Rotas relacionadas a usuários
│   └── index.js       # Arquivo principal para consolidar todas as rotas
├── services/          # Lógica de negócios e manipulação de dados
│   └── userService.js # Serviço para lógica de usuários
│   └── roomService.js # Serviço para lógica de salas
├── middlewares/       # Middlewares para validação, autenticação, etc.
│   └── auth.js        # Middleware para autenticação JWT
├── db/                # Configurações e migrações do banco de dados
│   ├── migrations/    # Arquivos de migração do Sequelize
│   └── config/        # Configurações do banco de dados
├── utils/             # Funções utilitárias
│   └── logger.js      # Função para logs
│   └── helpers.js     # Funções auxiliares
└── app.js             # Configuração principal do Express
└── server.js          # Inicialização do servidor
